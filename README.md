# pybrl ⠏⠽⠃⠗⠇

Braille (6-dot cell patterns) for Python

	import pybrl

	print pybrl.braille("braille for python is cool!") -> "⠃⠗⠁⠊⠇⠇⠑⠀⠋⠕⠗⠀⠏⠽⠞⠓⠕⠝⠀⠊⠎⠀⠉⠕⠕⠇⠮"

	print pybrl.ascii("⠃⠗⠁⠊⠇⠇⠑⠀⠋⠕⠗⠀⠏⠽⠞⠓⠕⠝⠀⠊⠎⠀⠉⠕⠕⠇⠮") -> "braille for python is cool!"
